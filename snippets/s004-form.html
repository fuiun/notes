<script type="text/javascript">
  function selectAll() {
    var items = document.getElementsByName('name1');
    var handle = document.getElementById('selectAll');
    if(handle.checked) {
      var i = 0;
      while(i<items.length) {
        items[i].checked = true;
        i++;
      }
    } else {
      var i = 0;
      while(i<items.length) {
        items[i].checked = false;
        i++;
      }
    }
  }
</script>
